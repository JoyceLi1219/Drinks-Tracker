<html lang="zh-TW">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>2026 的我喝了多少手搖</title>
    <!-- Tailwind CSS -->
    <script src="https://cdn.tailwindcss.com"></script>
    <!-- React & ReactDOM -->
    <script src="https://unpkg.com/react@18/umd/react.development.js"></script>
    <script src="https://unpkg.com/react-dom@18/umd/react-dom.development.js"></script>
    <!-- Babel for JSX -->
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <!-- Lucide Icons -->
    <script src="https://unpkg.com/lucide@latest"></script>
    <style>
        @import url('https://fonts.googleapis.com/css2?family=Noto+Sans+TC:wght@400;700;900&display=swap');
        body {
            font-family: 'Noto Sans TC', sans-serif;
            background-color: #D4EBF8;
        }
        .custom-dashed-border {
            border-style: dashed;
            border-width: 3px;
            border-color: rgba(169, 214, 241, 0.6);
        }
    </style>
</head>
<body>
    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect, useMemo } = React;
        
        // 模擬 Lucide 圖示組件
        const Icon = ({ name, size = 24, className = "" }) => {
            const [svg, setSvg] = useState("");
            useEffect(() => {
                if (window.lucide) {
                    const iconNode = window.lucide.icons[name];
                    if (iconNode) {
                        const svgString = iconNode.toSvg({
                            width: size,
                            height: size,
                            class: className,
                            'stroke-width': 2
                        });
                        setSvg(svgString);
                    }
                }
            }, [name, size, className]);
            return <span dangerouslySetInnerHTML={{ __html: svg }} />;
        };

        const App = () => {
            const colors = {
                piplupBlueLight: '#D4EBF8', 
                piplupBlueMedium: '#A9D6F1', 
                piplupBlueDark: '#3498DB', 
                piplupYellow: '#FEEFAD', 
                textMain: '#2C3E50',
            };

            const [activeTab, setActiveTab] = useState('daily');
            const [currentDate, setCurrentDate] = useState(new Date(2026, 0, 1));
            const [selectedDate, setSelectedDate] = useState(new Date(2026, 0, 1).toDateString());
            const [today, setToday] = useState(new Date().toDateString());
            
            const [records, setRecords] = useState(() => {
                const saved = localStorage.getItem('drink_records_2026');
                return saved ? JSON.parse(saved) : {};
            });

            const [editingDraft, setEditingDraft] = useState(null);

            const iceOptions = ["固定", "正常", "少冰", "微冰", "去冰", "常溫", "溫"];
            const sugarOptions = ["固定", "全糖", "少糖", "半糖", "微糖", "微微糖", "一分糖", "無糖"];

            useEffect(() => {
                localStorage.setItem('drink_records_2026', JSON.stringify(records));
            }, [records]);

            const stats = useMemo(() => {
                let monthlyCount = 0, monthlySpend = 0, yearlyCount = 0, yearlySpend = 0;
                const shopFreq = {}, itemFreq = {};
                const monthlyData = Array.from({ length: 12 }, () => ({ count: 0, spend: 0 }));
                const targetMonth = currentDate.getMonth();

                Object.entries(records).forEach(([dateStr, dayRecords]) => {
                    const d = new Date(dateStr);
                    if (d.getFullYear() === 2026) {
                        const count = dayRecords.length;
                        const spend = dayRecords.reduce((sum, r) => sum + Number(r.price || 0), 0);
                        const mIdx = d.getMonth();
                        yearlyCount += count;
                        yearlySpend += spend;
                        monthlyData[mIdx].count += count;
                        monthlyData[mIdx].spend += spend;
                        if (mIdx === targetMonth) {
                            monthlyCount += count;
                            monthlySpend += spend;
                        }
                        dayRecords.forEach(r => {
                            if (r.shop) shopFreq[r.shop] = (shopFreq[r.shop] || 0) + 1;
                            if (r.item) itemFreq[r.item] = (itemFreq[r.item] || 0) + 1;
                        });
                    }
                });

                const topShop = Object.entries(shopFreq).sort((a, b) => b[1] - a[1])[0]?.[0] || "尚未有資料";
                const topItem = Object.entries(itemFreq).sort((a, b) => b[1] - a[1])[0]?.[0] || "尚未有資料";
                return { monthlyCount, monthlySpend, yearlyCount, yearlySpend, topShop, topItem, monthlyData };
            }, [records, currentDate]);

            const daysInMonth = (y, m) => new Date(y, m + 1, 0).getDate();
            const firstDayOfMonth = (y, m) => new Date(y, m, 1).getDay();

            const monthData = useMemo(() => {
                const y = currentDate.getFullYear();
                const m = currentDate.getMonth();
                const total = daysInMonth(y, m);
                const start = firstDayOfMonth(y, m);
                const days = [];
                for (let i = 0; i < start; i++) days.push(null);
                for (let i = 1; i <= total; i++) days.push(new Date(y, m, i));
                return days;
            }, [currentDate]);

            const startNewDraft = () => {
                if ((records[selectedDate]?.length || 0) >= 2) return;
                setEditingDraft({ id: Date.now(), shop: "", item: "", ice: "正常", sugar: "半糖", price: "" });
            };

            const saveRecord = () => {
                if (!editingDraft?.shop || !editingDraft?.item) return;
                const dayRecords = records[selectedDate] || [];
                setRecords({ ...records, [selectedDate]: [...dayRecords, { ...editingDraft, price: Number(editingDraft.price) || 0 }] });
                setEditingDraft(null);
            };

            const deleteRecord = (id) => {
                const dayRecords = records[selectedDate].filter(r => r.id !== id);
                setRecords({ ...records, [selectedDate]: dayRecords });
            };

            const CloudIcon = () => (
                <svg width="30" height="20" viewBox="0 0 30 20" fill="none" className="opacity-30">
                    <path d="M5 14C5 11.2386 7.23858 9 10 9C10.1554 9 10.3088 9.00706 10.4598 9.02078C11.5312 6.13621 14.3216 4 17.6 4C21.466 4 24.6 7.13401 24.6 11C24.6 11.085 24.5985 11.1697 24.5954 11.254C26.5451 11.8596 28 13.6558 28 15.8C28 18.6719 25.6719 21 22.8 21H7.6C4.50721 21 2 18.4928 2 15.4C2 14.7397 2.11438 14.1061 2.32415 13.5186C3.04286 13.8291 3.99331 14 5 14Z" fill="white" stroke="#D4EBF8" strokeWidth="1"/>
                </svg>
            );

            const BarChart = ({ data, title, unit, colorClass }) => {
                const maxValue = Math.max(...data, 1);
                const months = ['1月', '2月', '3月', '4月', '5月', '6月', '7月', '8月', '9月', '10月', '11月', '12月'];
                return (
                    <div className="bg-white p-6 rounded-[32px] border-2 border-slate-100 shadow-sm">
                        <h4 className="text-sm font-black text-gray-400 uppercase tracking-widest mb-6 flex items-center gap-2">
                            {title} <span className="text-[10px] bg-blue-50 text-blue-500 px-2 py-0.5 rounded-full">{unit}</span>
                        </h4>
                        <div className="flex items-end justify-between h-48 gap-1">
                            {data.map((val, i) => (
                                <div key={i} className="flex-1 flex flex-col items-center group relative">
                                    <div className="absolute -top-10 scale-0 group-hover:scale-100 transition-transform bg-gray-800 text-white text-[10px] font-black px-2 py-1 rounded-lg z-20 pointer-events-none">{val}{unit}</div>
                                    <div className={`w-full rounded-t-lg transition-all duration-500 ${colorClass} ${val > 0 ? 'opacity-100' : 'opacity-10'}`} style={{ height: `${(val / maxValue) * 100}%`, minHeight: val > 0 ? '4px' : '0' }}></div>
                                    <span className="text-[9px] font-black text-gray-300 mt-2">{months[i]}</span>
                                </div>
                            ))}
                        </div>
                    </div>
                );
            };

            return (
                <div className="min-h-screen p-4 md:p-8">
                    <div className="max-w-5xl mx-auto bg-white rounded-[40px] shadow-2xl overflow-hidden border-2 border-white relative">
                        {/* 內層裝飾虛線 */}
                        <div className="absolute inset-2 custom-dashed-border rounded-[34px] pointer-events-none z-0"></div>
                        
                        {/* 裝飾雲朵 */}
                        <div className="absolute top-12 right-12"><CloudIcon /></div>
                        <div className="absolute top-48 left-8"><CloudIcon /></div>

                        <div className="relative z-10">
                            <header className="p-10 text-center border-b border-slate-100" style={{ backgroundColor: colors.piplupYellow }}>
                                <h1 className="text-2xl md:text-3xl font-black flex items-center justify-center gap-3">
                                    <div className="p-2 bg-white rounded-2xl shadow-sm"><Icon name="Coffee" className="text-blue-500" /></div>
                                    2026 的我喝了多少手搖
                                </h1>
                                <div className="flex justify-center mt-6">
                                    <div className="bg-white p-1.5 rounded-2xl flex gap-1 shadow-sm border border-yellow-200/50">
                                        <button onClick={() => setActiveTab('daily')} className={`flex items-center gap-2 px-6 py-2 rounded-xl text-sm font-black transition-all ${activeTab === 'daily' ? 'bg-blue-500 text-white' : 'text-blue-400'}`}>
                                            <Icon name="Calendar" size={16} /> 日常紀錄
                                        </button>
                                        <button onClick={() => setActiveTab('stats')} className={`flex items-center gap-2 px-6 py-2 rounded-xl text-sm font-black transition-all ${activeTab === 'stats' ? 'bg-blue-500 text-white' : 'text-blue-400'}`}>
                                            <Icon name="PieChart" size={16} /> 統計分析
                                        </button>
                                    </div>
                                </div>
                            </header>

                            {/* 數據看板 */}
                            <div className="grid grid-cols-2 lg:grid-cols-4 gap-4 p-6 bg-slate-50/30">
                                <div className="bg-white p-5 rounded-3xl border border-slate-100 text-center">
                                    <p className="text-[10px] text-gray-400 font-black uppercase tracking-widest">本月杯數</p>
                                    <p className="text-2xl font-black text-blue-600">{stats.monthlyCount} 杯</p>
                                </div>
                                <div className="bg-white p-5 rounded-3xl border border-slate-100 text-center">
                                    <p className="text-[10px] text-gray-400 font-black uppercase tracking-widest">本月花費</p>
                                    <p className="text-2xl font-black text-green-600">${stats.monthlySpend}</p>
                                </div>
                                <div className="bg-white p-5 rounded-3xl border border-slate-100 text-center">
                                    <p className="text-[10px] text-gray-400 font-black uppercase tracking-widest">年度總計</p>
                                    <p className="text-2xl font-black text-purple-600">{stats.yearlyCount} 杯</p>
                                </div>
                                <div className="bg-white p-5 rounded-3xl border border-slate-100 text-center">
                                    <p className="text-[10px] text-gray-400 font-black uppercase tracking-widest">年度總額</p>
                                    <p className="text-2xl font-black text-orange-600">${stats.yearlySpend}</p>
                                </div>
                            </div>

                            <div className="border-t border-slate-100 min-h-[600px]">
                                {activeTab === 'daily' ? (
                                    <div className="flex flex-col lg:flex-row">
                                        {/* 日曆區 */}
                                        <div className="w-full lg:w-1/2 p-8 lg:border-r border-slate-100 bg-white">
                                            <div className="flex items-center justify-between mb-8">
                                                <button onClick={() => setCurrentDate(new Date(currentDate.getFullYear(), currentDate.getMonth() - 1))} className="p-3 bg-slate-50 rounded-2xl text-blue-500"><Icon name="ChevronLeft" /></button>
                                                <h2 className="text-2xl font-black">{currentDate.getFullYear()} 年 {currentDate.getMonth() + 1} 月</h2>
                                                <button onClick={() => setCurrentDate(new Date(currentDate.getFullYear(), currentDate.getMonth() + 1))} className="p-3 bg-slate-50 rounded-2xl text-blue-500"><Icon name="ChevronRight" /></button>
                                            </div>
                                            <div className="grid grid-cols-7 gap-2 mb-4 text-center text-[10px] font-black text-slate-300">
                                                {['SUN', 'MON', 'TUE', 'WED', 'THU', 'FRI', 'SAT'].map(d => <div key={d}>{d}</div>)}
                                            </div>
                                            <div className="grid grid-cols-7 gap-3">
                                                {monthData.map((date, idx) => {
                                                    if (!date) return <div key={idx} />;
                                                    const dateStr = date.toDateString();
                                                    const count = records[dateStr]?.length || 0;
                                                    const isSelected = selectedDate === dateStr;
                                                    return (
                                                        <button 
                                                            key={dateStr}
                                                            onClick={() => {setSelectedDate(dateStr); setEditingDraft(null);}}
                                                            className={`aspect-square rounded-[18px] flex items-center justify-center text-sm font-black transition-all ${isSelected ? 'bg-blue-500 text-white scale-110 shadow-lg' : count > 0 ? 'bg-blue-100 text-blue-600' : 'bg-slate-50'}`}
                                                        >
                                                            {date.getDate()}
                                                        </button>
                                                    );
                                                })}
                                            </div>
                                            <div className="mt-12 space-y-4">
                                                <div className="bg-yellow-50/50 p-6 rounded-[28px] border-2 border-yellow-100/50">
                                                    <p className="text-[10px] font-black text-yellow-600 uppercase mb-1">最常消費門市</p>
                                                    <p className="text-xl font-black text-gray-700">{stats.topShop}</p>
                                                </div>
                                                <div className="bg-blue-50/50 p-6 rounded-[28px] border-2 border-blue-100/50">
                                                    <p className="text-[10px] font-black text-blue-600 uppercase mb-1">口袋愛喝品項</p>
                                                    <p className="text-xl font-black text-gray-700">{stats.topItem}</p>
                                                </div>
                                            </div>
                                        </div>

                                        {/* 列表區 */}
                                        <div className="w-full lg:w-1/2 p-8 bg-slate-50/30">
                                            <div className="flex justify-between items-center mb-8">
                                                <h3 className="text-xl font-black flex items-center gap-2">
                                                    <div className="w-2 h-8 bg-blue-500 rounded-full"></div>
                                                    {new Date(selectedDate).toLocaleDateString('zh-TW', {month:'long', day:'numeric'})}
                                                </h3>
                                                {!editingDraft && (
                                                    <button onClick={startNewDraft} disabled={(records[selectedDate]?.length || 0) >= 2} className="bg-blue-500 text-white px-6 py-3 rounded-2xl font-black disabled:opacity-30">新增</button>
                                                )}
                                            </div>

                                            {editingDraft && (
                                                <div className="bg-white p-7 rounded-[32px] shadow-xl border-2 border-blue-100 mb-6">
                                                    <div className="flex justify-between mb-4">
                                                        <span className="font-black">紀錄中...</span>
                                                        <button onClick={() => setEditingDraft(null)}><Icon name="X" className="text-slate-300" /></button>
                                                    </div>
                                                    <input placeholder="店家" className="w-full bg-slate-50 rounded-2xl p-4 mb-3 font-bold outline-none" onChange={e => setEditingDraft({...editingDraft, shop: e.target.value})} />
                                                    <input placeholder="品項" className="w-full bg-slate-50 rounded-2xl p-4 mb-4 font-bold outline-none" onChange={e => setEditingDraft({...editingDraft, item: e.target.value})} />
                                                    <div className="flex flex-wrap gap-2 mb-4">
                                                        {iceOptions.map(o => <button key={o} onClick={()=>setEditingDraft({...editingDraft, ice:o})} className={`px-3 py-1 text-xs rounded-lg border-2 ${editingDraft.ice === o ? 'bg-blue-500 text-white border-blue-500' : 'bg-white border-slate-50'}`}>{o}</button>)}
                                                    </div>
                                                    <div className="flex flex-wrap gap-2 mb-6">
                                                        {sugarOptions.map(o => <button key={o} onClick={()=>setEditingDraft({...editingDraft, sugar:o})} className={`px-3 py-1 text-xs rounded-lg border-2 ${editingDraft.sugar === o ? 'bg-blue-500 text-white border-blue-500' : 'bg-white border-slate-50'}`}>{o}</button>)}
                                                    </div>
                                                    <input type="number" placeholder="金額" className="w-full bg-slate-50 rounded-2xl p-4 mb-6 font-black outline-none" onChange={e => setEditingDraft({...editingDraft, price: e.target.value})} />
                                                    <button onClick={saveRecord} className="w-full bg-blue-500 text-white py-4 rounded-2xl font-black">儲存紀錄</button>
                                                </div>
                                            )}

                                            <div className="space-y-4">
                                                {records[selectedDate]?.map(r => (
                                                    <div key={r.id} className="bg-white p-6 rounded-[32px] border border-slate-100 flex justify-between items-center shadow-sm">
                                                        <div>
                                                            <p className="text-[10px] text-blue-300 font-black">{r.shop}</p>
                                                            <p className="text-xl font-black text-slate-700">{r.item}</p>
                                                            <div className="flex gap-2 mt-2">
                                                                <span className="text-[9px] bg-slate-50 px-2 py-1 rounded text-slate-400 font-bold">{r.ice} / {r.sugar}</span>
                                                            </div>
                                                        </div>
                                                        <div className="text-right">
                                                            <p className="text-2xl font-black text-blue-500">${r.price}</p>
                                                            <button onClick={() => deleteRecord(r.id)} className="text-slate-200 hover:text-red-400 transition-colors mt-2"><Icon name="Trash2" size={16} /></button>
                                                        </div>
                                                    </div>
                                                ))}
                                            </div>
                                        </div>
                                    </div>
                                ) : (
                                    <div className="p-8 space-y-8 bg-white">
                                        <div className="grid grid-cols-1 md:grid-cols-2 gap-8">
                                            <BarChart title="每月飲用杯數" unit="杯" colorClass="bg-blue-400" data={stats.monthlyData.map(m => m.count)} />
                                            <BarChart title="每月消費金額" unit="元" colorClass="bg-green-400" data={stats.monthlyData.map(m => m.spend)} />
                                        </div>
                                    </div>
                                )}
                            </div>
                        </div>
                    </div>
                    <footer className="mt-8 text-center text-[9px] font-black opacity-20 uppercase tracking-widest">
                        Stay Hydrated • 2026 Personal Beverage Archive
                    </footer>
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
</body>
